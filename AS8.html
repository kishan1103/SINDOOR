import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load the Titanic dataset
titanic = sns.load_dataset('titanic')

# Replace numeric values in 'survived' with string labels for clarity
titanic['survived'] = titanic['survived'].map({0: 'No', 1: 'Yes'})

# Plot 1: Count of survivors and non-survivors
sns.countplot(x='survived', data=titanic)
plt.title("Survival Count")
plt.xlabel("Survived")
plt.ylabel("Number of Passengers")
plt.show()

# Plot 2: Survival by gender
sns.countplot(x='sex', hue='survived', data=titanic)
plt.title("Survival by Gender")
plt.xlabel("Gender")
plt.ylabel("Number of Passengers")
plt.legend(title="Survived")
plt.show()

# Plot 3: Age distribution
sns.histplot(titanic['age'].dropna(), kde=True, bins=30)
plt.title("Age Distribution")
plt.xlabel("Age")
plt.ylabel("Frequency")
plt.show()

# Plot 4: Survival by class
sns.countplot(x='class', hue='survived', data=titanic)
plt.title("Survival by Passenger Class")
plt.xlabel("Class")
plt.ylabel("Number of Passengers")
plt.legend(title="Survived")
plt.show()

# Plot 5: Fare distribution
sns.boxplot(x='survived', y='fare', data=titanic)
plt.title("Fare Distribution by Survival")
plt.xlabel("Survived")
plt.ylabel("Fare")
plt.show()

# Plot 6: Heatmap of missing values
sns.heatmap(titanic.isnull(), cbar=False, cmap='viridis')
plt.title("Missing Values in Titanic Dataset")
plt.show()
